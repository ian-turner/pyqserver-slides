\section{Introduction}

\subsection{Quantum Programming Languages}

\begin{frame}{Quantum Programming Languages}
  \begin{itemize}
    \item Quantum programming languages allow programmers to design quantum algorithms at varying levels of abstraction
    \item Low-level: OpenQASM~\cite{cross2022openqasm} --- an intermediate representation for describing quantum circuits
    \item Mid-level: Cirq~\cite{cirq} and Qiskit~\cite{qiskit} --- use Python libraries to construct quantum circuits
    \item High-level: Silq~\cite{bichsel2020silq}, Q\#~\cite{svore2018q}, Quipper~\cite{green2013quipper}, Proto-Quipper~\cite{fu2023proto} --- provide advanced features such as automatic uncomputation, type-checking, dynamic lifting, and type-safe circuit reversing and control
  \end{itemize}
\end{frame}

\subsection{Proto-Quipper}

\begin{frame}{Proto-Quipper}
  \begin{itemize}
    \item Proto-Quipper is a family of functional quantum programming languages giving a formal foundation to Quipper~\cite{fu2023proto,fu2024proto}
    \item Has two runtimes: \emph{circuit generation time} and \emph{circuit execution time}
    \begin{itemize}
      \item Variables fixed at generation time are called \emph{parameters}; variables fixed at execution time are called \emph{states}
    \end{itemize}
    \item \emph{Dynamic lifting} allows state variables to be promoted to parameters --- measurement results from execution time can inform generation of the next circuit
    \item This interleaving of the two runtimes enables more complex circuits, but is expensive: all active qubits must be held in long-term storage while awaiting classical control~\cite{fu2023proto}
  \end{itemize}
\end{frame}

\subsection{Simulating Proto-Quipper}

\begin{frame}{Simulating Proto-Quipper}
  \begin{itemize}
    \item Proto-Quipper uses a client-server architecture for simulation~\cite{fu2023proto}
    \item The \emph{client} is the Proto-Quipper interpreter; the \emph{server} is the simulator
    \item The client sends quantum instructions (QServer commands~\cite{fu2023proto}) to the server; the server sends back measurement results via dynamic lifting
  \end{itemize}

\subsection{Contributions}

\begin{frame}{Contributions}
  \begin{itemize}
    \item Extend the Proto-Quipper simulation architecture with state-of-the-art simulators from Cirq and Qiskit, enabling dynamic lifting on third-party simulators
    \item Optimize the new simulators via \emph{instruction queuing} --- deferring instructions that do not need to be evaluated immediately
    \item Demonstrate that the new simulators significantly outperform the existing custom Haskell simulator across a variety of benchmarks
  \end{itemize}
\end{frame}

\section{Integrating Third-Party Simulators}

\subsection{Available Simulators}

\begin{frame}{Available Simulators}
  \begin{itemize}
    \item Dozens of state-of-the-art quantum simulators exist, employing a variety of simulation methods (statevector, tensor networks, decision trees, etc.)~\cite{jamadagni2024benchmarking}
    \item We focus on \emph{statevector simulators}, which model quantum states via matrix-vector multiplication
    \item We chose Cirq and Qiskit: the most widely used and actively developed quantum computing libraries, each with multiple state-of-the-art simulation backends
    \item Both support \emph{dynamic circuits} (classically-controlled gates, while loops), but these are experimental and insufficient for dynamic lifting: they do not allow communication between the Python runtime and the circuit simulation runtime
  \end{itemize}
\end{frame}

\subsection{Integration with Proto-Quipper}

\begin{frame}{Integration with Proto-Quipper}
  \begin{itemize}
    \item Key challenges of integrating Cirq and Qiskit with Proto-Quipper:
    \begin{itemize}
      \item No dynamic qubit allocation --- ancilla qubits must be pre-allocated at circuit generation time
      \item A circuit object must be constructed for every simulation, even single-gate instructions
      \item Measurement semantics differ: Proto-Quipper measurements \emph{consume} a qubit; Cirq/Qiskit leave qubits in a collapsed state
    \end{itemize}
    \item Solution: use OpenQASM as an intermediate representation --- both simulators have OpenQASM import functions, and QServer is syntactically similar to OpenQASM 2
  \end{itemize}
\end{frame}

\subsection{Simulation with Qiskit}

\begin{frame}{Simulation with Qiskit}
  \begin{itemize}
    \item QServer has four operation types; each is translated for Qiskit (and similarly for Cirq):
    \begin{itemize}
      \item \textbf{Initialization}: new qubit tensored onto existing statevector ($|\psi'\rangle = |0\rangle \otimes |\psi\rangle$); X gate applied if initializing to $|1\rangle$
      \item \textbf{Unitary gates}: current statevector re-initialized into a new circuit, gate applied, updated statevector extracted
      \item \textbf{Measurements}: built-in simulator function collapses statevector and returns result; measured qubit then removed from statevector
      \item \textbf{Read command}: instruction queue flushed and executed, measurement performed, collapsed state saved in memory, result returned to client
    \end{itemize}
  \end{itemize}
\end{frame}

\subsection{Instruction Queuing}

\begin{frame}{Instruction Queuing}
  \begin{itemize}
    \item The server only returns results on \texttt{read} commands, so all other instructions can be deferred
    \item Instructions are kept in a queue and evaluated as one circuit when a \texttt{read} command arrives
    \item Advantages over instruction-by-instruction execution:
    \begin{itemize}
      \item Statevector initialized only once per queue flush instead of before each gate
      \item Enables gate fusion and circuit rewriting optimizations within Cirq/Qiskit~\cite{elsman2025gate}
    \end{itemize}
  \end{itemize}
\end{frame}

\subsection{Converting to OpenQASM}

\begin{frame}[fragile]{Converting to OpenQASM}
  \begin{itemize}
    \item QServer commands are parsed as received; in queuing mode, non-measurement commands are placed in a queue
    \item On a \texttt{read} command, queued commands are converted to OpenQASM via simple pattern matching and run by the simulator
  \end{itemize}
  \begin{center}
  \begin{minipage}{0.7\textwidth}
  \begin{lstlisting}[language=Python, morekeywords={case,match}, basicstyle=\ttfamily\scriptsize]
case Q():
    idx = self.free_qubits.pop()
    self.qubit_map[command.reg] = idx
    gate_str = 'reset qs[%d];' % idx
    if command.bvalue:
        gate_str += '\nx qs[%d];' % idx
    return gate_str
case CNOT():
    x = self.qubit_map[command.x]
    y = self.qubit_map[command.y]
    return 'cx qs[%d], qs[%d];' % (x, y)
  \end{lstlisting}
  \end{minipage}
  \end{center}
\end{frame}

bell00 : !(Unit -> Qubit * Qubit)
bell00 x =
  let a = H (Init0 ())
      b = Init0 ()
      (b, a) = CNot b a
  in (a, b)

bellMeas : Bool * Bool
bellMeas =
  let (a, b) = bell00 ()
  in (dynlift (Meas a),

\section{Benchmark Results}

\begin{frame}{Benchmarks Overview}
  We evaluate simulators on a suite of Proto-Quipper benchmark programs:
  \begin{itemize}
    \item Quantum Fourier Transform (QFT)
    \item Binary adders: QFT adder, Cuccaro adder, Gidney adder~\cite{ruiz2017quantum,cuccaro2004new,gidney2018halving}
    \item MaxCut QAOA~\cite{farhi2014quantum}
    \item Multi-qubit teleportation~\cite{pirandola2015advances}
    \item Repeat-Until-Success (RUS) circuits~\cite{paetznick2013repeat}
  \end{itemize}
  \vspace{0.5em}
  Each benchmark compares: \textbf{Haskell server}, \textbf{queued} and \textbf{non-queued} Python server (Cirq/Qiskit), and \textbf{direct} Qiskit/Cirq (no Proto-Quipper).
\end{frame}
