\section{Introduction}

\subsection{Quantum Programming Languages}

\begin{frame}{Quantum Programming Languages}
  \begin{itemize}
    \item Quantum programming languages allow programmers to design quantum algorithms at varying levels of abstraction
    \item Low-level: OpenQASM --- an intermediate representation for describing quantum circuits
    \item Mid-level: Cirq and Qiskit --- use Python libraries to construct quantum circuits
    \item High-level: Silq, Q\#, Quipper, Proto-Quipper --- provide advanced features such as automatic uncomputation, type-checking, dynamic lifting, and type-safe circuit reversing and control
  \end{itemize}
\end{frame}

\subsection{Proto-Quipper}

\begin{frame}{Proto-Quipper}
  \begin{itemize}
    \item Proto-Quipper is a family of functional quantum programming languages giving a formal foundation to Quipper
    \item Has two runtimes: \emph{circuit generation time} and \emph{circuit execution time}
    \begin{itemize}
      \item Variables fixed at generation time are called \emph{parameters}; variables fixed at execution time are called \emph{states}
    \end{itemize}
    \item \emph{Dynamic lifting} allows state variables to be promoted to parameters --- measurement results from execution time can inform generation of the next circuit
    \item This interleaving of the two runtimes enables more complex circuits, but is expensive: all active qubits must be held in long-term storage while awaiting classical control
  \end{itemize}
\end{frame}

\subsection{Simulating Proto-Quipper}

\begin{frame}{Simulating Proto-Quipper}
  \begin{itemize}
    \item Proto-Quipper uses a client-server architecture for simulation
    \item The \emph{client} is the Proto-Quipper interpreter; the \emph{server} is the simulator
    \item The client sends quantum instructions (QServer commands) to the server; the server sends back measurement results via dynamic lifting
  \end{itemize}

\subsection{Contributions}

\begin{frame}{Contributions}
  \begin{itemize}
    \item Extend the Proto-Quipper simulation architecture with state-of-the-art simulators from Cirq and Qiskit, enabling dynamic lifting on third-party simulators
    \item Optimize the new simulators via \emph{instruction queuing} --- deferring instructions that do not need to be evaluated immediately
    \item Demonstrate that the new simulators significantly outperform the existing custom Haskell simulator across a variety of benchmarks
  \end{itemize}
\end{frame}
